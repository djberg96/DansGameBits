\scriptsize\textbf{Railroads}

\begin{enumerate}\setlength{\itemsep}{0.15em}\setlength{\parskip}{0pt}\setlength{\topsep}{0.2em}
\item The Virginia Central Railroad can be used only by the Confederates. It can carry up to 20 Stacking Factors per Turn.
\item The B\&O Railroad can be used only by the Union. It can carry up to 30 Stacking Factors per Turn.
\item No other railroad may be used by either side.
\end{enumerate}
\end{minipage}%
}

\begin{minipage}[t]{0.485\textwidth}
\fbox{%
\begin{minipage}[t]{0.95\linewidth}
\scriptsize\textbf{Rallying Troops}

\begin{enumerate}\setlength{\itemsep}{0.15em}\setlength{\parskip}{0pt}\setlength{\topsep}{0.2em}
\item General must be present in hex.
\item Player wishing to rally troops rolls one die.
\item If number rolled is less than or equal to the general's value, the `D' result can be ignored.
\item If a `6' is rolled, the general's unit counter is eliminated.
\end{enumerate}
\end{minipage}%
}
