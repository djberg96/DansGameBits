\textbf{Strength Modifications}
\begin{itemize}
\item $\times \frac{1}{2}$ if a unit entered through an adjacent bridge or river hexside this turn.
\item $\times \frac{1}{2}$ if combat occurs in a ridge hex and the unit entered from a non-road hex.
\item $\times 2$ for a unit defending in an entrenchment.
\item The attacker shifts one column left if the defender is entrenched.
\end{itemize}

\textbf{Attacker}
\begin{itemize}
\item +1 if no attacker moved during the turn
\item -2 if any attacker force marched during the turn
\item +1 if the attacker chooses \textit{Pitched Battle}
\item -1 if the attacker chooses \textit{Probe}
\end{itemize}

\textbf{Defender}
\begin{itemize}
\item +2 if no defender moved during the turn
\item -1 if any defender attacked while entrained
\item +3 if any defender is entrenched
\item +1 if the defender chooses \textit{Pitched Battle}
\item -1 if the defender chooses \textit{Withdrawal}
\item -1 if any defender is demoralized
\end{itemize}

\textbf{Command Rating Modifier}
\begin{itemize}
\item +1 if 3 or more combat units are in the stack and the highest-ranking leader is not a department or corps leader
