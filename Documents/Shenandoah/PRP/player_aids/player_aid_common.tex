\newcommand{\sequencepanel}{%
  \begin{tikzpicture}
    \node[draw=panelrule,rounded corners=4pt,fill=white,inner sep=10pt,text width=\textwidth-24pt] (panel) {%
      {\Large\bfseries Sequence of Play}\par\vspace{0.25em}
      \small
      \setlength{\multicolsep}{0pt}
      \setlength{\columnsep}{18pt}
      \begin{multicols}{2}
      \setlength{\parskip}{0.18em}
      \textbf{I. Initial Phase}\par
      \begin{itemize}
      \item Reinforcements
      \item Replacements
      \item Wagon Creation
      \end{itemize}
      \vspace{0.26em}
      \textbf{II. Initiative Phase}\par
      Determine initiative for the turn.\par

      \vspace{0.26em}
      \textbf{III. First Command Phase}\par
      \begin{itemize}
      \item Receive command points
      \item Broadcast command points
      \end{itemize}

      \vspace{0.26em}
      \textbf{IV. First Movement Phase}\par
      \begin{itemize}
      \item Move and receive reinforcements
      \item Opponent may react
      \item Destroy/repair bridges or rail; entrench
      \end{itemize}

      \columnbreak

      \textbf{V. Second Command Phase}\par
      Same as the first command phase, for the second player.\par

      \vspace{0.26em}
      \textbf{VI. Second Movement Phase}\par
      Same as the first movement phase, for the second player.\par

      \vspace{0.26em}
      \textbf{VII. Combat Phase}\par
      \begin{itemize}
      \item Combat
      \item Depletion
      \end{itemize}

      \vspace{0.26em}
      \textbf{VIII. Conclusion Phase}\par
      \begin{itemize}
      \item Morale Recovery
      \item Replenishment
      \item Union Panic Track (1862 only)
      \end{itemize}
      \end{multicols}
    };
  \end{tikzpicture}%
}
